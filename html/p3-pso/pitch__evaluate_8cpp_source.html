132 
133  num_voiced = num_unvoiced = num_voiced_unvoiced = num_unvoiced_voiced = 0;
134  num_voiced_voiced = num_gross_errors = 0;
135  fine_error = 0.0F;
136  int nfine = 0;
137 
138  if (vref.size() != vtest.size())
139  return;
140 
141  for (unsigned int i = 0; i < vref.size(); ++i) {
142 
143  if (vref[i] == 0.0F)
144  num_unvoiced++;
145  else
146  num_voiced++;
147 
148  if (vref[i] == 0.0F and vtest[i] == 0.0F)
149  continue;
150 
151  if (vref[i] == 0.0F and vtest[i] != 0.0F) {
152  num_unvoiced_voiced++;
153  } else if (vref[i] != 0.0F and vtest[i] == 0.0F) {
154  num_voiced_unvoiced++;
155  } else {
156  float f = fabs((vref[i] - vtest[i])/vref[i]);
157  num_voiced_voiced++;
158  if (f > gross_threshold) {
159  num_gross_errors++;
160  } else {
161  nfine++;
162  fine_error += f*f;
163  }
164  }
165  }
166  if (nfine > 0)
167  fine_error = sqrt(fine_error/nfine);
