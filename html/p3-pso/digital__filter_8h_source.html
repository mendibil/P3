17 /**
18  The circular index is a simple class to address a circular buffer.
19  Once the buffer size is defined, you can add or substract and the index
20  remains inside the buffer. A casting operator to int is provided.
21 
22  Example:
23 ~~~~~~~~~~~~~~~{.c}
24  float x[8]={0,1,2,4,5,6,7};
25  CircularIndex ci(8);
26  int n=10; float z=0.0F;
27  for (n=0; n< 10; ++n, ++ci)
28  z += x[ci+1] - x[ci-1];
29 ~~~~~~~~~~~~~~~
30  Operations +, +=, ++, -, -=, -- are defined
31 **/
32  class CircularIndex {
33  int index, ///the index (value)
34  size; ///buffer size; index goes from 0 ... size-1
35  public:
36  CircularIndex(int s=1): index(0), size(s) {} ///< default: size=1; not very useful (index can only be 0)
37  CircularIndex(CircularIndex &ci): index(ci.index), size(ci.size) {}///< constructor defining size based on other CircularIndex
38  void resize(int s) {size=s; index=0;} ///<Set size of circular index
39  CircularIndex & operator++() {if (++index == size) index = 0; return *this;}///<Increment
40  CircularIndex & operator+=(int i) {index += i; while (index >= size) index -= size; return *this;}///<Increment 'i' positions
41  CircularIndex & operator+(int i) {return CircularIndex(*this) += i;}///<Increment 'i' positions
42  CircularIndex & operator--() {if (index == 0) index = size; --index; return *this;}///<Decrement
43  CircularIndex & operator-=(int i) {index -= i; while (index < 0) index += size; return *this;}///<Decrement 'i' positions
44  CircularIndex & operator-(int i) {return CircularIndex(*this) -= i;}///<Decrement 'i' positions
45  operator int() {return index;}///!Casting from CircularIndex to int, returning int.
46  CircularIndex & operator= (const CircularIndex &ci) {index = ci.index; size = ci.size; return *this;}///!Assignment operator.
47  };

50  /**
51  This class implements digital filtering.
52  The filter is stored as rational funcion H(z) = g B(z)/A(z)
53  Being
54  g: gain
55  B(z) = b0 +b1 z^-1 + b2 z^-2 ... bm z^-m
56  A(z) = a0 +a1 z^-1 + a2 z^-2 ... an z^-n
57 
58  The state conditions are saved after each call to the filter.
59  The filter is implemented using the direct form.
60  Y(z) = B(z)/A(z) X(z) => U(z) = X(z)/A(z); Y(z) = B(z) U(z)
61  State conditions: u[n]
62 
63  The state conditions are only cleared either:
64  - explicitely calling clean()
65  - by a change in filter coeficients that imply an order filter
66 
67  The filters coefficients are set using set_a(), set_b(), set_gain().
68  An alternative convinient method exist for 2nd order resonators,
69  using the central frequency and the resonator bandwidth: set_resonator()
70  **/
